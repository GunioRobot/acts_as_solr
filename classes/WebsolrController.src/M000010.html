  def start
    if @user || @pass
      if(COMMANDS.include?(@command))
        send("cmd_#{@command}")
      else
        puts @parser
        exit(1)
      end
    else
      puts "\nYou need to specify your username and password, either on the command\nline with the -u and -p flags, or in the WEBSOLR_USER and WEBSOLR_PWD\nenvironment variables.\n\n"
      exit(1)
    end
  end
