  def initialize(parser)
    @options = parser.options
    @command = @options.delete(:command)
    @parser = parser
    @user = @options[:user] || ENV["WEBSOLR_USER"]
    @pass = @options[:pass] || ENV["WEBSOLR_PWD"]
    @base = "http://#{URI::escape @user}:#{URI::escape @pass}@websolr.com"
  end
